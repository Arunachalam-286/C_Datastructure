#include<stdio.h>
#include<stdbool.h>

bool isPalindrome(int x) {
    // If the number is negative or ends with 0 but is not 0 itself, it cannot be a palindrome
    if (x < 0 || (x % 10 == 0 && x != 0)) {
        return false;
    }

    int reversed = 0;
    int original = x;

    // Reverse the number until all digits are processed
    while (x > 0) {
        int digit = x % 10;      // Get the last digit
        reversed = reversed * 10 + digit; // Build the reversed number
        x /= 10;                 // Remove the last digit from x
    }

    // Check if the original number equals the reversed number
    return original == reversed;
}

int main() {
    int x;
    printf("Enter a number: ");
    scanf("%d", &x);

    if (isPalindrome(x)) {
        printf("True\n");
    } else {
        printf("False\n");
    }

    return 0;
}
